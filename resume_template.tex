\newcommand{\makeheader}{
	\begin{tabu*}{@{} l @{}}
		%% NAME %%
		{\fontsize{30pt}{36pt}\selectfont \name} \\
		%% Street Address %%
		\streetaddress \\
		%% City, State zip
		\city \\
		%% email address %%
		\email \\
		%% phone number %%
		\phonenumber \\
	\end{tabu*}
}
\newcommand{\makesubjecttitle}[1]{
	{\fontsize{20pt}{20pt}\selectfont #1}
}
\newcommand{\makeentrywithdate}[3]{
	\begin{tabu*}{@{} L{\entrytitlewidthwithdate} R{\entrydatewidth} @{}}
		{\fontsize{14pt}{17pt}\selectfont #1} & #2 \\
		\multicolumn{2}{L{\entrywidth}}{#3}
	\end{tabu*}
}
\newcommand{\makeentrywithoutdate}[2]{
	\begin{tabu*}{@{} L{\entrywidth} @{}}
		{\fontsize{14pt}{17pt}\selectfont #1} \\
		#2
	\end{tabu*}
}
